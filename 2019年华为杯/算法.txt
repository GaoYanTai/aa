a={}          #定义一个空字典，用于存储怠速片段的起始位置和片段长度
for x in range(0,file_num):            #遍历文件，file_num为文件个数
    打开每个文件           
    for i in range(sh.shape[0]):   #遍历文件每一行数据，sh.shape[0]为行数
        if GPS速度小于10:
            a[i]=1      #初始化从i开始的怠速片段长度为1
            for j in range(i+1,sh.shape[0]):#从i行继续遍历该文件的每一行数据
                if sh[j][1]<10:  #如果i后为连续怠速条件，字典a中键i对应的值为怠速时长
                    a[i]+=1
                else:         #如果出现非怠速条件，跳出该循环
                    break
            if a[i]>180:   #判断怠速时长是否超过三分钟
                取字典中的键i，
                取字典中键对应的值a[i]，
                for 从i到i+a[i]行的数据：
                    GPS速度=0
                    油门踏板开度=0
                    其余各项指标=从i到i+a[i]行对应数据的平均值
                    将更改后信息写入文件
            else:
                 for 从i到i+a[i]行的数据：
			 将原信息写入文件
        else:
            不小于10的行数据直接写入文件
